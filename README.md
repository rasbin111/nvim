# Keymaps

leader = <spacebar>

## netrw file manager
:Exp - file explorer 
d - create a directory 
% - create a file 

move a file: 
    - mt -> target directory 
    - mf -> moving file / folder 
    - mm -> execute move 

# lsp 
gd - go to defintion 
gD - go to declaration 
gi - go to implementation 
gl - diagnostic 

# telescope 
leader ff - find files 
leader fg - live grep 
leader fb - find buffers 
leader fh - find help


# basics 
:w - write 
:q - quit
